        // Функция для получения случайной цитаты из массива
        function getRandomQuote() {
            // Генерируем случайный индекс от 0 до (длина массива - 1)
            const randomIndex = Math.floor(Math.random() * quotes.length);
            // Возвращаем объект цитаты по этому индексу
            return quotes[randomIndex];
        }

        // Функция для отображения цитаты на странице
        function displayQuote() {
            // Получаем случайную цитату
            const quote = getRandomQuote();
            // Устанавливаем текст цитаты в соответствующий HTML-элемент
            quoteTextElement.textContent = quote.text;
            // Устанавливаем автора цитаты в соответствующий HTML-элемент
            quoteAuthorElement.textContent = quote.author;
        }
